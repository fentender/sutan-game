# 如何修改或新建一个仪式（rite）
### 仪式（rite）说明
- 通常意义上的事件，有专门的界面可以放置卡牌进行操作
![图片](https://docimg4.docs.qq.com/image/AgAABS6GRke1tB6LSDpLPqYWQX5MNM-p.png?w=679&h=444)
### 如何新建一个仪式（rite）
- 在mod提供的地址中新建未被使用的数字50开头的7位数id的json文件
    - 补充说明：50开头的7位数id是官方推荐的id段，制作者也可以使用任意开头未被使用的7位数
    只要不和已有的id重复就行。
![图片](https://docimg7.docs.qq.com/image/AgAABS6GRkfPl6Nb1r5Fb5Bizi03SGWn.png?w=572&h=268)
### rite的数据结构说明
```json
{
        "id": 5000019, //未被使用的数字50开头的7位数id
        "name": "圣训六十篇",//仪式标题
        "text": "阅读可以提升自己，养成阅读的习惯吧",  //仪式启动界面正文    
        "tips": "",//策划看的备注
        "once_new":0, 
        //用来设置仪式出现在桌面上图标的表现
        //配置为0：每次出现在桌面上时先显示【?】图标，用来让玩家理解是先刷出的仪式
        //配置为1：只有在第1次出现时显示【?】图标。下次再刷新会正常显示仪式自身的图标
        //配置为-1：每次出现都只显示仪式自身的图标
        "round_number": 1, 
        //仪式执行的回合数
        //配置为0代表仪式执行后立即结算，不用过回合
        "waiting_round":1,
        //仪式在执行前，在桌面停留的回合数，回合数一过默认仪式会被移除
        //配置为0代表仪式在执行前会一直停留在桌面
        "waiting_round_end_action":[ 
	        //仪式在停留时间结束时会立即执行的动作，可以不配置
	        //详细见范例1
        ],
        "method_settlement": "", //参数要保留，但没有用，不用配置
        "auto_begin": 0, 
        //用来配置仪式过回合时是否会自动执行
        //配置1为会自动执行，配置为0代表不会自动执行
        "auto_result":0,
        //用来配置是否显示跳过仪式结算界面的开关
        //配置为1显示，配置为0不显示
        "location": "自宅:[2,12]", 
        //仪式生成时在桌面上的坐标位置
        //可以从已有的坐标中选择配置
        //目前暂不开放新坐标的编辑
        "icon": "rite_7",  
        //仪式的图标，配置对应图标的资源名。对应资源的放置在mod要求的对应文件地址中
        "tag_tips": [
	        //参数要保留，但没有用，不用配置
        ],
        "tag_tips_up":{
	        //用来提示玩家仪式内有哪些标签会参加结果检定
        	//如果没有检定内容的话可以不配置
        	//详细见范例1
        },
        "tips_text":[
	        //仪式的提示文本
        	//如果提示文本的话可以不配置
        ],
        "open_conditions": [
           //仪式额外需要满足的启动执行的条件
           //详细见范例1
        ],
        "random_text":{
	        //参数要保留，但没有用，不用配置
        },
        "random_text_up":{
	        //创建骰子检定的编号，和对应的提示文本
        	//如果没有骰子检定的话可以不配置
        	//详细见范例1
        },
        "settlement_prior": [
	        //仪式结算的结果在这里配置
        	//需要当前仪式优先检定结算的结果在这里配置
        	//只会执行其中一个最先满足条件的结果
	        //执行了这里的结果，settlement和settlement_extre里的结果将不再结算
        	//详细剑范例1
        ],                
        "settlement": [
            //仪式结算的结果在这里配置
        	//只会执行其中一个最先满足条件的结果
	        //以下为一个结果配置的标准格式
            {
                "condition":
                {
	                //执行这个结果需要满足的条件
                	//根据策划需求选用可用的条件
		            //根据策划需求可以配置多个，并灵活使用any，all等语法
		            //如果配置为空则视作满足条件
                },
                "result_title":"", //结果的标题文字
                "result_text": "学习到了社交技巧",//结果的正文
                "result":
                {   
                    //根据策划需求选用可用的动作，玩家可见大部分动作生效的表现
                    //根据策划需求可以配置多个，并灵活使用choose，delay等语法  
                    //在这里不能执行【生成仪式】和【激活或关闭幕后】的动作 
                    //执行的动作只有仪式所有结果全部结算后才会生效
                    //意味着你不能在这个结果里设置一个计数器，立即就在下一个结果里区检定它
                    //动作的执行顺序在下方action里配置的动作之前                          
                },
                "action":
                {
	                //根据策划需求选用可用的动作，一般可见动作生效的表现
                    //根据策划需求可以配置多个，并灵活使用choose，delay等语法  
                    //在这里不能对当前仪式卡牌进行操作，比如"s1+体魄":1，这类时不行的
                    //动作的执行顺序在下方result里配置的动作之后
                }
            }

        ],                
        "settlement_extre": [
            //仪式结算的结果在这里配置
        	//如果没有执行settlement_prior，此处的结果只要满足条件都会结算
        	//结算顺序从上至下
        ],
        "cards_slot": //仪式界面上的槽位配置，玩家可以操作的部分，也是仪式检定的主体
        {
            "s1": //槽位1
                {
                    "condition": 
                    {          
	                    //需要放置满足配置条件的卡牌，不满足的卡牌放不上去
	                	//根据策划需求选用可用的条件
			            //根据策划需求可以配置多个，并灵活使用any，all等语法
			            //如果配置为空则视作任意卡都可以放入   
                    },
                    "open_adsorb":1, 
                    //用来设置当前槽位是否自动吸附卡牌
                    //配置为1时，仪式生成时会自动吸取满足条件的卡牌
                    //配置为0关闭自动吸附
                    "is_key":0, //参数要保留，但没有用，不用配置
                    "is_empty":0,
                    //用来设置当前槽位是否可以为空
                    //配置为1时，当前槽位必须放入满足条件的卡牌，否则仪式无法生成和执行
                    //配置为0时，当前槽位可以不放卡牌
                    "is_enemy":1,
                    //用来设置当前槽位的卡牌是否时敌人，在骰子检定公式里可以区分敌我
                    //配置为1时是敌人
                    //配置为0时是友方
                    "text":"正在阅读的书" //当前槽位的提示文本
                },
            "s2": //槽位2
                {
                    "condition": 
                    {
                    },
                    "open_adsorb":0,
                    "is_key":0,
                    "is_empty":0,
                    "is_enemy":0,
                    "text":"阅读者",
                    "pops":[ 
                    //置入卡牌时，当前槽位显示对应的气泡，用于增强互动感。可以不配置
                        {
                            "condition":{
                                //执行这个气泡需要满足的条件
                            },
                            "action":{
                               "pop.5000019_s2_01.self":"他这些漂亮话说得真好听啊！"
                               //执行对应的气泡指令
                               //详细见【命令（动作）】中的说明
                            }
                        }
                    ]
                }
        }
    }
```
